# Pull every .c file from the cloned doomgeneric core, EXCEPT files that
# would drag in dependencies we don't have on this badge:
#   - doomgeneric_*.c   : alternate platform layers (SDL, Allegro, X11,
#                          Linuxvt, Emscripten, Soso, Win32). Our own
#                          main/doomgeneric_esp32c6.c replaces them.
#   - i_allegro*, i_sdl*, i_oplmusic, mus2mid : audio backends. With
#     FEATURE_SOUND undefined (we don't define it), i_sound.c compiles
#     to no-ops referencing nothing — so we can drop every audio backend
#     wholesale. No music, no SFX, but DOOM runs.
#   - i_cdmus.c         : CD audio (no CD).
#   - i_endoom.c        : terminal "ENDOOM" graphic on quit (no terminal).
#   - i_joystick.c      : SDL joystick (no SDL).
set(DG_SRC_DIR "${CMAKE_SOURCE_DIR}/doomgeneric/doomgeneric")
file(GLOB DG_SOURCES "${DG_SRC_DIR}/*.c")
list(FILTER DG_SOURCES EXCLUDE REGEX "doomgeneric_[a-z0-9]+\\.c$|i_allegro.+\\.c$|i_sdl.+\\.c$|i_oplmusic\\.c$|i_cdmus\\.c$|i_endoom\\.c$|i_joystick\\.c$|mus2mid\\.c$")

idf_component_register(
    SRCS
        "main.c"
        "doomgeneric_esp32c6.c"
        "display.c"
        "buttons.c"
        "led_feedback.c"
        "doom_stubs.c"
        "wad_mmap.c"
        "baked_textures.c"   # generated by tools/bake_textures.py
        ${DG_SOURCES}
    INCLUDE_DIRS
        "."
        "${DG_SRC_DIR}"
    PRIV_REQUIRES
        esp_driver_spi
        esp_driver_gpio
        esp_driver_rmt
        esp_lcd
        esp_timer
        nvs_flash
        esp_partition
        vfs
        button
        led_strip
)

# DOOM screen resolution → 320x200 (native DOOM) so a single LCD blit
# fits inside the panel. CMAP256 uses 1 byte/pixel framebuffer (64 KB)
# instead of the default 4-byte ARGB (256 KB) — critical for fitting in
# the ESP32-C6's 512 KB SRAM.
target_compile_definitions(${COMPONENT_LIB} PRIVATE
    DOOMGENERIC_RESX=320
    DOOMGENERIC_RESY=200
    CMAP256
    NORMALUNIX           # tells doom it's a unix-y platform (we satisfy fopen etc.)
    LINUX                # same
)

# DOOM compiles loose with lots of legacy warnings — silence them.
# Several of these are GCC's "warnings as errors" defaults under ESP-IDF
# that DOOM's 1993-era code freely violates (char-subscripts in isspace,
# pointer-sign in string ops, missing-field-initializers in event_t).
# Wrap I_ZoneBase so doom_stubs.c can override DOOM's "want 6 MiB or die"
# autosize loop with our tiered-fallback malloc strategy. The linker
# redirects all callers of I_ZoneBase to __wrap_I_ZoneBase; the upstream
# version becomes __real_I_ZoneBase (we don't use it).
target_link_options(${COMPONENT_LIB} INTERFACE
    "-Wl,--wrap=I_ZoneBase"
    # W_OpenFile is wrapped so wad_mmap.c returns a flash-mmap'd
    # wad_file_t instead of going through the upstream class array.
    "-Wl,--wrap=W_OpenFile"
    # D_FindIWAD is wrapped because upstream stats real filesystem paths
    # to find a WAD — we have no filesystem, just a raw flash partition.
    # Our wrap returns a constant filename + hardcoded GameMission_t.
    "-Wl,--wrap=D_FindIWAD"
)

target_compile_options(${COMPONENT_LIB} PRIVATE
    -Wno-char-subscripts
    -Wno-pointer-sign
    -Wno-unused-but-set-variable
    -Wno-unused-but-set-parameter
    -Wno-unused-variable
    -Wno-unused-function
    -Wno-unused-result
    -Wno-unused-parameter
    -Wno-format
    -Wno-implicit-fallthrough
    -Wno-array-bounds
    -Wno-stringop-truncation
    -Wno-stringop-overflow
    -Wno-restrict
    -Wno-misleading-indentation
    -Wno-missing-field-initializers
    -Wno-discarded-qualifiers
    -Wno-int-conversion
    -Wno-incompatible-pointer-types
    -Wno-deprecated-declarations
    -Wno-absolute-value
    -Wno-shift-negative-value
    -Wno-sign-compare
    -Wno-maybe-uninitialized
    -Wno-dangling-pointer
    -Wno-array-parameter
    -Wno-stringop-overread
    -Wno-aggressive-loop-optimizations
)
